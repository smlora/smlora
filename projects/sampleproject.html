def rPCA(D, lam=0.01, tau=1.1, rho=1.1, tol=1e-6, maxiter=500, plot=True):
    L = np.zeros(D.shape)
    S = np.zeros(D.shape)
    Y = D.copy()*D.max()/lam
    mu = tau/np.linalg.norm(Y, 2)
    mu_bar = mu/tol
    d_norm = np.linalg.norm(D, 'fro')
    for i in range(maxiter):
        X = D - L + (Y/mu)
        S = np.sign(X)*np.maximum(np.abs(X)-lam/mu, 0)
        u, s, v = np.linalg.svd(D-S+Y/mu, full_matrices=False)
        svp = (s > 1/mu).sum()
        L = (u[:, :svp]*(s[:svp]-1/mu))@v[:svp]
        Y = Y + mu*(D-L-S)
        mu = min(mu*rho, mu_bar)
        if np.linalg.norm(D-L-S, 'fro')/d_norm < tol: break

    prompt = 'Algorithm converged in %s iterations.' % (i+1)\
                if i+1 < maxiter else 'Algorithm did not converged.'
    print(prompt)
    if plot:
        plt.figure(figsize=(10, 4))
        plt.subplot(131), plt.axis('off'), plt.title('Original Image')
        plt.imshow(D, cmap='gray', vmin=D.min(), vmax=D.max())
        plt.subplot(132), plt.axis('off'), plt.title('Low-rank Image')
        plt.imshow(L, cmap='gray', vmin=D.min(), vmax=D.max())
        plt.subplot(133), plt.axis('off'), plt.title('Sparse Image')
        plt.imshow(S, cmap='gray', vmin=D.min(), vmax=D.max())
        plt.tight_layout()
        plt.show()
    return L, S
